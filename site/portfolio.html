AuthController.php
public function login(Request $req): JsonResponse
{
  $user = User::findByEmail($req->email);

  if (!$user->verifyPassword($req->password)) {
    return response()->json([
      'error' => 'Unauthorized'
    ], 401);
  }

  $token = JWT::encode([
    'sub' => $user->id,
    'exp' => now()->addHours(8)
  ]);

  return response()->json([
    'token' => $token
  ]);
}
